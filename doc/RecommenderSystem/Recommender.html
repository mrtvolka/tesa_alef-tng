def self.relations
  if @@rels.empty?
    @@rels = UserToLoRelation.where('user_id = (?) AND learning_object_id IN (?)',
                                   @@user_id,
                                   nontest_learning_objects.map(&:id)
    ).order(:created_at)
  end
  @@rels
end

def normalize list

  max = list.values.max

  unless max == 0
    list.each do |key,val|
      list[key] = val.to_f / max
    end
  end

  list

end
